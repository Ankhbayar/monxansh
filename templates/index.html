JSON format for jQuery

jQuery.ajax({
  url : '//monxansh.appspot.com/xansh.json?currency=USD|EUR',
  data_type: 'json',
  success : function(data) {
    jQuery.each(data, function(index, row) {
      // Format
      // row.code       - Код USD, EUR...
      // row.name       - Нэр
      // row.rate       - Ханш
      // row.rate_float - Ханш
      // row.rate_date  - Огноо
      // row.last_date  - Мэдээний татсан огноо
      alert(row.code);
  });
}
});
